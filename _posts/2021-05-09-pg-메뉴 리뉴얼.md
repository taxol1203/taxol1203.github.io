## 코드  
```java
import java.util.*;
import java.util.Map.Entry;

class Solution {
    static Map<String, Integer> menu;
	static char sel[];
	 
	static int n, size;
    public static String[] solution(String[] orders, int[] course) {
        String[] answer = {};
        ArrayList<String> result = new ArrayList<>();
        for(int len : course) {
        	
        	menu = new HashMap<>();
        	size = len;
        	sel = new char[size];
        	for(String str : orders) {
        		n = str.length();
        		DFS(0,0,str);
        	}
        	
        	List<Entry<String, Integer>> list = new ArrayList<>(menu.entrySet());  // 중요!
        	Collections.sort(list, (o1,o2)->{
        		return -o1.getValue().compareTo(o2.getValue()); // 중요!
        	});
        	
        	if(list.isEmpty() || list.get(0).getValue() < 2)
        		continue;
        	
        	int maxN = list.get(0).getValue();

        	for(Entry<String, Integer> en : list) {
        		if(en.getValue() != maxN)
        			break;
        		result.add(en.getKey());
        	}
        }
        
        answer = new String[result.size()];
        for(int i = 0; i < result.size(); i++) {
        	answer[i] = result.get(i);
        }
        Arrays.sort(answer);
        return answer;
    }
    
    static void DFS(int lv, int start, String order) {
    	if(lv == size) {
    		
    		char[] cl = sel.clone();
    		Arrays.sort(cl);
    		String str = String.valueOf(cl);	// 중요!
    		
    		if(menu.containsKey(str))
    			menu.put(str, menu.get(str) + 1);
    		else
    			menu.put(str, 1);
    		
    		return;
    	}
    	
    	for(int i = start; i < n; i++) {
    		sel[lv] = order.charAt(i);
    		DFS(lv + 1, i + 1, order);
    	}
    }
}
```
